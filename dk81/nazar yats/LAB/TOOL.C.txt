
#include "tool.h"

int stage(int inNumeric, int inStage)
{
	int numeric = 1;
	int i;

	for ( i = 0; i < inStage; i++)
	{
		numeric = numeric  * inNumeric;
	}

	return numeric;
}

int factorial(int inNumeric)
{
	int numeric = 1;
	int i;

	for ( i = 0; i < inNumeric; i++)
	{
		numeric = numeric * (inNumeric - i);
	}

	return numeric;
}

double operation(int a, int b, int c)
{
	double result = ((double)(stage(a, 2) + stage(b, 2)) / ()(b + stage(5, c))) + factorial(b);
	return result;
}
